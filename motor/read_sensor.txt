#include <Wire.h>

void setup()
{
  Wire.begin();
  Serial.begin(9600);
  for(int i=0; i<16; ++i) {
    pinMode(i, OUTPUT);
  }
}


void loop()
{
  delay(1000);
  Wire.requestFrom(0x10, 1);
  while(Wire.available()) {
    int pin = Wire.read();
    Serial.println(pin);
  }
  for(int i=0; i<8; ++i) {
    int val = analogRead(i);
    Serial.print(val);
    Serial.print(" ");
  }
  Serial.println();
}
